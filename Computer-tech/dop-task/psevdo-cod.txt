// Объявление переменных
цел n, i, maxIndex
вещ A[0..99], maxAbs, sum
симв FNAME, FNAME_RES
цел MAX_SIZE

MAX_SIZE = 100
FNAME = "tests/test1.txt"
FNAME_RES = "result.txt"

// Работа с файлами
ОТКРЫТЬ файл FNAME для чтения

// Проверка открытия входного файла
ЕСЛИ файл не открылся
    ТО
    ВЫВЕСТИ "Ошибка: не удалось открыть файл ", FNAME
    КОНЕЦ
КОНЕЦ ЕСЛИ

// Проверка, что файл не пустой
ЕСЛИ файл пустой
    ТО
    ВЫВЕСТИ "Ошибка: файл пустой"
    ЗАКРЫТЬ файл
    КОНЕЦ
КОНЕЦ ЕСЛИ

// Чтение размера массива
ПРОЧЕСТЬ n из файла
ЕСЛИ ошибка чтения n
    ТО
    ВЫВЕСТИ "Ошибка: некорректные данные о размере массива"
    ЗАКРЫТЬ файл
    КОНЕЦ
КОНЕЦ ЕСЛИ

// Проверка размера массива
ЕСЛИ n < 1 ИЛИ n > MAX_SIZE
    ТО
    ВЫВЕСТИ "Ошибка: размер массива должен быть от 1 до ", MAX_SIZE
    ЗАКРЫТЬ файл
    КОНЕЦ
КОНЕЦ ЕСЛИ

// Выделение памяти и чтение элементов массива
ВЫДЕЛИТЬ память для массива A размером n
i = 0
ПОКА i < n
НЦ
    ПРОЧЕСТЬ A[i] из файла
    ЕСЛИ ошибка чтения A[i]
        ТО
        ЕСЛИ конец файла
            ТО
            ВЫВЕСТИ "Ошибка: недостаточно элементов в файле"
        ИНАЧЕ
            ВЫВЕСТИ "Ошибка: некорректные элементы массива"
        КОНЕЦ ЕСЛИ
        ОСВОБОДИТЬ память массива A
        ЗАКРЫТЬ файл
        КОНЕЦ
    КОНЕЦ ЕСЛИ
    i = i + 1
КЦ

ЗАКРЫТЬ файл

// Вывод исходного массива
ВЫВЕСТИ "Исходный массив: "
i = 0
ПОКА i < n
НЦ
    ВЫВЕСТИ A[i], " "
    i = i + 1
КЦ
ВЫВЕСТИ перевод строки

// Поиск индекса максимального по модулю элемента
maxIndex = 0
maxAbs = |A[0]|
i = 1
ПОКА i < n
НЦ
    ЕСЛИ |A[i]| > maxAbs
        ТО
        maxAbs = |A[i]|
        maxIndex = i
    КОНЕЦ ЕСЛИ
    i = i + 1
КЦ

ВЫВЕСТИ "Максимальный по модулю элемент: ", A[maxIndex], " (индекс ", maxIndex, ")"

// Вычисление суммы элементов до максимального по модулю
sum = 0.0
i = 0
ПОКА i < maxIndex
НЦ
    sum = sum + A[i]
    i = i + 1
КЦ

ВЫВЕСТИ "Сумма элементов до максимального по модулю: ", sum

// Вывод элементов после максимального по модулю в обратном порядке
ВЫВЕСТИ "Элементы после максимального по модулю (в обратном порядке): "
i = n - 1
ПОКА i > maxIndex
НЦ
    ВЫВЕСТИ A[i], " "
    i = i - 1
КЦ
ВЫВЕСТИ перевод строки

// Запись результатов в файл
ОТКРЫТЬ файл FNAME_RES для записи

ЗАПИСАТЬ "Сумма элементов до максимального по модулю: ", sum
ЗАПИСАТЬ "Элементы после максимального по модулю (в обратном порядке): "
i = n - 1
ПОКА i > maxIndex
НЦ
    ЗАПИСАТЬ A[i], " "
    i = i - 1
КЦ
ЗАПИСАТЬ перевод строки

ЗАКРЫТЬ файл

ВЫВЕСТИ "Результаты записаны в файл ", FNAME_RES

// Освобождение памяти
ОСВОБОДИТЬ память массива A

КОНЕЦ
