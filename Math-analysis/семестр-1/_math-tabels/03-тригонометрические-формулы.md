# Тригонометрические формулы

## Основные тождества

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Равно} \\
\hline
\hline
\sin^2 x + \cos^2 x & 1 \\
\hline
1+\tan^2 x & \dfrac{1}{\cos^2 x} \\
\hline
1+\cot^2 x & \dfrac{1}{\sin^2 x} \\
\hline
\end{array}
$$

## Чётность и периодичность

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Равно} \\
\hline
\hline
\sin(-x) & -\sin x \\
\hline
\cos(-x) & \cos x \\
\hline
\tan(-x) & -\tan x \\
\hline
\sin(x+2\pi) & \sin x \\
\hline
\cos(x+2\pi) & \cos x \\
\hline
\tan(x+\pi) & \tan x \\
\hline
\end{array}
$$

## Формулы сложения

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Равно} \\
\hline
\hline
\sin(a\pm b) & \sin a\cos b \pm \cos a\sin b \\
\hline
\cos(a\pm b) & \cos a\cos b \mp \sin a\sin b \\
\hline
\tan(a\pm b) & \dfrac{\tan a \pm \tan b}{1 \mp \tan a\tan b} \\
\hline
\end{array}
$$

## Двойной угол

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Равно} \\
\hline
\hline
\sin 2x & 2\sin x\cos x \\
\hline
\cos 2x & \cos^2 x-\sin^2 x \\
\hline
\cos 2x & 1-2\sin^2 x \\
\hline
\cos 2x & 2\cos^2 x-1 \\
\hline
\tan 2x & \dfrac{2\tan x}{1-\tan^2 x} \\
\hline
\end{array}
$$

## Половинный угол

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Равно} \\
\hline
\hline
\sin^2\dfrac{x}{2} & \dfrac{1-\cos x}{2} \\
\hline
\cos^2\dfrac{x}{2} & \dfrac{1+\cos x}{2} \\
\hline
\tan\dfrac{x}{2} & \dfrac{1-\cos x}{\sin x} \\
\hline
\tan\dfrac{x}{2} & \dfrac{\sin x}{1+\cos x} \\
\hline
\end{array}
$$

## Произведение в сумму

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Равно} \\
\hline
\hline
\sin a\sin b & \dfrac{\cos(a-b)-\cos(a+b)}{2} \\
\hline
\sin a\cos b & \dfrac{\sin(a+b)+\sin(a-b)}{2} \\
\hline
\cos a\cos b & \dfrac{\cos(a-b)+\cos(a+b)}{2} \\
\hline
\end{array}
$$

## Сумма в произведение

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Равно} \\
\hline
\hline
\sin a + \sin b & 2\sin\dfrac{a+b}{2}\cos\dfrac{a-b}{2} \\
\hline
\sin a - \sin b & 2\cos\dfrac{a+b}{2}\sin\dfrac{a-b}{2} \\
\hline
\cos a + \cos b & 2\cos\dfrac{a+b}{2}\cos\dfrac{a-b}{2} \\
\hline
\cos a - \cos b & -2\sin\dfrac{a+b}{2}\sin\dfrac{a-b}{2} \\
\hline
\end{array}
$$
