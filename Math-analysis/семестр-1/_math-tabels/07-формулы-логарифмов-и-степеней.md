# Формулы логарифмов и степеней

## Степени

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Значение} \\
\hline
\hline
a^0 & 1 \\
\hline
a^{-x} & \dfrac{1}{a^x} \\
\hline
a^{x+y} & a^x a^y \\
\hline
a^{x-y} & \dfrac{a^x}{a^y} \\
\hline
(a^x)^y & a^{xy} \\
\hline
(ab)^x & a^x b^x \\
\hline
a^x & e^{x\ln a} \\
\hline
\end{array}
$$

## Логарифмы

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Формула} & \text{Значение} \\
\hline
\hline
\ln 1 & 0 \\
\hline
\ln e & 1 \\
\hline
\ln(ab) & \ln a + \ln b \\
\hline
\ln(a/b) & \ln a - \ln b \\
\hline
\ln(a^k) & k\ln a \\
\hline
\log_a b & \dfrac{\ln b}{\ln a} \\
\hline
\log_a a & 1 \\
\hline
\log_a 1 & 0 \\
\hline
\log_a b & \dfrac{1}{\log_b a} \\
\hline
\end{array}
$$
