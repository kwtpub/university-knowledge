# Таблица производных

## Правила дифференцирования

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Правило} & \text{Формула} \\
\hline
\hline
\text{Сумма/разность} & \displaystyle (f \pm g)' = f' \pm g' \\
\hline
\text{Произведение} & \displaystyle (fg)' = f'g + fg' \\
\hline
\text{Частное} & \displaystyle \left(\frac{f}{g}\right)' = \frac{f'g - fg'}{g^2} \\
\hline
\text{Сложная функция} & \displaystyle (f \circ g)' = f'(g(x))\,g'(x) \\
\hline
\end{array}
$$

## Базовые производные

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Функция} & \text{Производная} \\
\hline
\hline
C & 0 \\
\hline
x^n & n x^{n-1} \\
\hline
(ax+b)^n & n a (ax+b)^{n-1} \\
\hline
\sqrt{x} & \dfrac{1}{2\sqrt{x}} \\
\hline
\dfrac{1}{x} & -\dfrac{1}{x^2} \\
\hline
\dfrac{1}{ax+b} & -\dfrac{a}{(ax+b)^2} \\
\hline
\ln\lvert x\rvert & \dfrac{1}{x} \\
\hline
\ln\lvert ax+b\rvert & \dfrac{a}{ax+b} \\
\hline
\log_a x & \dfrac{1}{x\ln a} \\
\hline
e^x & e^x \\
\hline
e^{u(x)} & e^{u(x)} u'(x) \\
\hline
a^x & a^x \ln a \\
\hline
a^{u(x)} & a^{u(x)} \ln a\, u'(x) \\
\hline
\sin x & \cos x \\
\hline
\cos x & -\sin x \\
\hline
\tan x & \dfrac{1}{\cos^2 x} \\
\hline
\cot x & -\dfrac{1}{\sin^2 x} \\
\hline
\sec x & \sec x\,\tan x \\
\hline
\csc x & -\csc x\,\cot x \\
\hline
\arcsin x & \dfrac{1}{\sqrt{1-x^2}} \\
\hline
\arccos x & -\dfrac{1}{\sqrt{1-x^2}} \\
\hline
\arctan x & \dfrac{1}{1+x^2} \\
\hline
\arccot x & -\dfrac{1}{1+x^2} \\
\hline
\sinh x & \cosh x \\
\hline
\cosh x & \sinh x \\
\hline
\tanh x & \dfrac{1}{\cosh^2 x} \\
\hline
\coth x & -\dfrac{1}{\sinh^2 x} \\
\hline
\operatorname{arsinh} x & \dfrac{1}{\sqrt{x^2+1}} \\
\hline
\operatorname{arcosh} x & \dfrac{1}{\sqrt{x^2-1}} \\
\hline
\operatorname{artanh} x & \dfrac{1}{1-x^2} \\
\hline
\end{array}
$$
