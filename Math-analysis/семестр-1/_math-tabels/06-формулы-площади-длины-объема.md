# Формулы площади, длины и объёма

## Площадь

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Случай} & \text{Формула} \\
\hline
\hline
\text{Под графиком } y=f(x),\ f\ge 0 & S = \int_a^b f(x)\,dx \\
\hline
\text{Между } y=f(x)\ \text{и}\ y=g(x) & S = \int_a^b (f-g)\,dx \\
\hline
\text{Полярный сектор } r=r(\theta) & S = \dfrac{1}{2}\int_\alpha^\beta r(\theta)^2\,d\theta \\
\hline
\text{Между } r_1(\theta)\ \text{и}\ r_2(\theta) & S = \dfrac{1}{2}\int_\alpha^\beta (r_2^2-r_1^2)\,d\theta \\
\hline
\end{array}
$$

## Длина кривой

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Задание кривой} & \text{Формула} \\
\hline
\hline
y=f(x) & L = \int_a^b \sqrt{1+(f')^2}\,dx \\
\hline
x=x(t),\ y=y(t) & L = \int_\alpha^\beta \sqrt{(x')^2+(y')^2}\,dt \\
\hline
r=r(\theta) & L = \int_\alpha^\beta \sqrt{r^2+(r')^2}\,d\theta \\
\hline
\end{array}
$$

## Объём и поверхность вращения

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Случай} & \text{Формула} \\
\hline
\hline
\text{Объём по площади сечения} & V = \int_a^b A(x)\,dx \\
\hline
\text{Диски/шайбы вокруг } O_x & V = \pi\int_a^b (R^2-r^2)\,dx \\
\hline
\text{Диски/шайбы вокруг } O_y & V = \pi\int_c^d (R^2-r^2)\,dy \\
\hline
\text{Оболочки вокруг } O_y & V = 2\pi\int_a^b x f(x)\,dx \\
\hline
\text{Оболочки вокруг } O_x & V = 2\pi\int_c^d y g(y)\,dy \\
\hline
\text{Поверхность вокруг } O_x & S = 2\pi\int_a^b f(x)\sqrt{1+(f')^2}\,dx \\
\hline
\text{Поверхность вокруг } O_y & S = 2\pi\int_c^d g(y)\sqrt{1+(g')^2}\,dy \\
\hline
\end{array}
$$
