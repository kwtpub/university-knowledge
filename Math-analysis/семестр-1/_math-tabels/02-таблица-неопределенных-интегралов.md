# Таблица неопределенных интегралов

## Правила

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Правило} & \text{Формула} \\
\hline
\hline
\text{Линейность} & \displaystyle \int (\alpha f + \beta g)\,dx = \alpha\int f\,dx + \beta\int g\,dx \\
\hline
\text{Подстановка} & \displaystyle \int f(g(x)) g'(x)\,dx = \int f(u)\,du \\
\hline
\text{По частям} & \displaystyle \int u\,dv = uv - \int v\,du \\
\hline
\end{array}
$$

## Базовые интегралы

$$
\Large
\begin{array}{|c|c|}
\hline
\text{Функция} & \text{Интеграл} \\
\hline
\hline
k & kx + C \\
\hline
x^n,\ (n\ne -1) & \dfrac{x^{n+1}}{n+1} + C \\
\hline
(ax+b)^n,\ (n\ne -1) & \dfrac{(ax+b)^{n+1}}{a(n+1)} + C \\
\hline
\dfrac{1}{x} & \ln\lvert x\rvert + C \\
\hline
\dfrac{1}{ax+b} & \dfrac{1}{a}\ln\lvert ax+b\rvert + C \\
\hline
e^x & e^x + C \\
\hline
e^{ax} & \dfrac{e^{ax}}{a} + C \\
\hline
a^x & \dfrac{a^x}{\ln a} + C \\
\hline
\sin x & -\cos x + C \\
\hline
\cos x & \sin x + C \\
\hline
\tan x & -\ln\lvert \cos x\rvert + C \\
\hline
\cot x & \ln\lvert \sin x\rvert + C \\
\hline
\sec^2 x & \tan x + C \\
\hline
\csc^2 x & -\cot x + C \\
\hline
\sec x\tan x & \sec x + C \\
\hline
\csc x\cot x & -\csc x + C \\
\hline
\sec x & \ln\lvert \sec x + \tan x\rvert + C \\
\hline
\csc x & \ln\lvert \csc x - \cot x\rvert + C \\
\hline
\dfrac{1}{1+x^2} & \arctan x + C \\
\hline
\dfrac{1}{x^2+a^2} & \dfrac{1}{a}\arctan\dfrac{x}{a} + C \\
\hline
\dfrac{1}{a^2-x^2} & \dfrac{1}{2a}\ln\left\lvert \dfrac{a+x}{a-x}\right\rvert + C \\
\hline
\dfrac{1}{x^2-a^2} & \dfrac{1}{2a}\ln\left\lvert \dfrac{x-a}{x+a}\right\rvert + C \\
\hline
\dfrac{1}{\sqrt{a^2-x^2}} & \arcsin\dfrac{x}{a} + C \\
\hline
\dfrac{1}{\sqrt{x^2+a^2}} & \operatorname{arsinh}\dfrac{x}{a} + C \\
\hline
\dfrac{1}{\sqrt{x^2-a^2}} & \ln\left\lvert x+\sqrt{x^2-a^2}\right\rvert + C \\
\hline
\dfrac{1}{1-x^2} & \operatorname{artanh} x + C \\
\hline
\sinh x & \cosh x + C \\
\hline
\cosh x & \sinh x + C \\
\hline
\dfrac{1}{\cosh^2 x} & \tanh x + C \\
\hline
\dfrac{1}{\sinh^2 x} & -\coth x + C \\
\hline
\end{array}
$$
